// Submission pipeline
async evaluate(code, lang) {
  const job = await queue.add({
    code, lang,
    timeout: 2000,
    memLimit: '128m',
    syscalls: 'restricted'
  });

  const map = {
    'AC': 'Accepted ✓',
    'WA': 'Wrong Answer ✗',

  };
  dispatch(setVerdict(map[res]));
});
